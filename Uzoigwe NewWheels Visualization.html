#Reading the "Question3.csv" dataset into a dataframe (i.e.loading the data)
path="/content/drive/My Drive/Question3.csv"
df = pd.read_csv(path)

# Pivot the DataFrame to get customer_feedback categories as columns
pivot_df = df.pivot(index='quarter_number', columns='customer_feedback', values='percentage')

# Normalize the data to make it 100% stacked
pivot_df = pivot_df.divide(pivot_df.sum(axis=1), axis=0) * 100

# Create the 100% stacked bar chart
ax = pivot_df.plot(kind='bar', stacked=True, colormap='viridis', figsize=(10, 6))

# Add labels and title
ax.set_xlabel('Quarter Number')
ax.set_ylabel('Percentage')
ax.set_title('100% Stacked Bar Chart of Customer Feedback by Quarter')

# Add a legend
ax.legend(title='Feedback', bbox_to_anchor=(1.05, 1), loc='upper left')

# Add labels with percentages to the stacked bars
for container in ax.containers:
    ax.bar_label(container, label_type='edge', fontsize=10, labels=[f"{val:.2f}%" for val in container.datavalues])

# Show the plot
plt.tight_layout()
plt.show()
